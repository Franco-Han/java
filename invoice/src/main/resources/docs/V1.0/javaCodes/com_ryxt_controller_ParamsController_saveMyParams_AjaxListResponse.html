public class AjaxListResponse implements Serializable{

    private boolean success; //是否成功
    private int statusCode; //状态代码
    private String message; //返回消息
    private ParamsMy[] data; //数据

    public boolean getSuccess(){
        return success;
    }

    public void setSuccess(boolean success){
        this.success = success;
    }

    public int getStatusCode(){
        return statusCode;
    }

    public void setStatusCode(int statusCode){
        this.statusCode = statusCode;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public ParamsMy[] getData(){
        return data;
    }

    public void setData(ParamsMy[] data){
        this.data = data;
    }

}

public class ParamsMy implements Serializable{

    private String id; //
    private String userId; //
    private String paramsId; //

    public String getId(){
        return id;
    }

    public void setId(String id){
        this.id = id;
    }

    public String getUserId(){
        return userId;
    }

    public void setUserId(String userId){
        this.userId = userId;
    }

    public String getParamsId(){
        return paramsId;
    }

    public void setParamsId(String paramsId){
        this.paramsId = paramsId;
    }

}
