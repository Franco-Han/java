public class AjaxListResponse implements Serializable{

    private boolean success; //是否成功
    private int statusCode; //状态代码
    private String message; //返回消息
    private InvoiceDate[] data; //数据

    public boolean getSuccess(){
        return success;
    }

    public void setSuccess(boolean success){
        this.success = success;
    }

    public int getStatusCode(){
        return statusCode;
    }

    public void setStatusCode(int statusCode){
        this.statusCode = statusCode;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public InvoiceDate[] getData(){
        return data;
    }

    public void setData(InvoiceDate[] data){
        this.data = data;
    }

}

public class InvoiceDate implements Serializable{

    private String id; //ID
    private String startDate; //开始日期
    private String endDate; //结束日期
    private String userId; //用户id
    private String createDate; //创建日期
    private String deleteFlag; //是否删除

    public String getId(){
        return id;
    }

    public void setId(String id){
        this.id = id;
    }

    public String getStartDate(){
        return startDate;
    }

    public void setStartDate(String startDate){
        this.startDate = startDate;
    }

    public String getEndDate(){
        return endDate;
    }

    public void setEndDate(String endDate){
        this.endDate = endDate;
    }

    public String getUserId(){
        return userId;
    }

    public void setUserId(String userId){
        this.userId = userId;
    }

    public String getCreateDate(){
        return createDate;
    }

    public void setCreateDate(String createDate){
        this.createDate = createDate;
    }

    public String getDeleteFlag(){
        return deleteFlag;
    }

    public void setDeleteFlag(String deleteFlag){
        this.deleteFlag = deleteFlag;
    }

}
